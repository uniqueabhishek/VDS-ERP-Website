
model Vendor {
  id            String   @id @default(cuid())
  name          String   @unique
  contactPerson String?
  phone         String?
  email         String?
  address       String?
  gstNumber     String?
  notes         String?

  // Relations
  createdBy     String
  user          User     @relation(fields: [createdBy], references: [id], onDelete: Cascade)
  expenses      Expense[]

  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  @@map("vendors")
}
